[docs]def check_tensorised_shapes(tensorised: Sequence) -> None:
    aa = iter(tensorised)
    a = next(aa).shape[:-1]
    try:
        while True:
            b = next(aa).shape[:-1]
            if a != b:
                warn(f"{a},{b}, Does not match!")
            a = b
    except StopIteration:
        pass
